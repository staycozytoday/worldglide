**Step 1: Build Himalayas API scraper**

Create `src/lib/scrapers/api-himalayas.ts`. Himalayas API: `GET https://himalayas.app/jobs/api?limit=50&offset=0` — paginated, returns `{ jobs: [...], total_count }`. Each job has `locationRestrictions` (empty array = worldwide).

```typescript
import { Job, getCompanyLogoUrl } from "../types";
import { isWorldwideRemote } from "../filter";
import { categorizeJob } from "../categorize";
import { createJobId } from "../utils";
import { fetchWithRetry } from "../fetch-retry";

export interface HimalayasResult {
  jobs: Job[];
  rawCount: number;
}

export async function scrapeHimalayas(): Promise<HimalayasResult> {
  const allJobs: Job[] = [];
  let rawCount = 0;
  let offset = 0;
  const limit = 50;
  let totalPages = 0;
  const maxPages = 100; // Safety cap: 5,000 jobs max

  console.log("[himalayas] starting API fetch...");

  while (totalPages < maxPages) {
    const res = await fetchWithRetry(
      `https://himalayas.app/jobs/api?limit=${limit}&offset=${offset}`,
      { headers: { "User-Agent": "worldglide-jobs/1.0" }, timeoutMs: 30000 }
    );

    if (!res.ok) {
      console.error(`[himalayas] API returned ${res.status} at offset ${offset}`);
      break;
    }

    const data = await res.json();
    const items = data.jobs || [];
    if (items.length === 0) break;

    rawCount += items.length;

    for (const item of items) {
      // Build location string from their data
      const location = item.location || "";
      const locationRestrictions: string[] = item.locationRestrictions || [];

      if (
        !isWorldwideRemote({
          title: item.title,
          description: item.excerpt || item.description || "",
          location,
          locationRestrictions: locationRestrictions.length > 0 ? locationRestrictions : undefined,
        })
      ) {
        continue;
      }

      const category = categorizeJob(item.title);
      if (!category) continue;

      const companyDomain = item.companyUrl
        ? new URL(item.companyUrl).hostname.replace("www.", "")
        : undefined;

      allJobs.push({
        id: createJobId("himalayas", `${item.id || item.slug}`),
        title: item.title || "",
        company: item.companyName || "Unknown",
        companyLogo: companyDomain ? getCompanyLogoUrl(companyDomain) : undefined,
        category,
        url: item.applicationUrl || item.url || `https://himalayas.app/jobs/${item.slug}`,
        source: "himalayas",
        tags: item.categories || [],
        postedAt: item.pubDate || item.postedAt || new Date().toISOString(),
        scrapedAt: new Date().toISOString(),
        description: (item.excerpt || "").slice(0, 200),
        isWorldwide: true,
        employmentType: item.type || "Full-time",
      });
    }

    offset += limit;
    totalPages++;

    // Rate limit: 500ms between pages
    await new Promise((r) => setTimeout(r, 500));
  }

  console.log(`[himalayas] ${allJobs.length} worldwide jobs from ${rawCount} raw`);
  return { jobs: allJobs, rawCount };
}
```

**Step 2: Build RemoteOK API scraper**

Create `src/lib/scrapers/api-remoteok.ts`. RemoteOK API: `GET https://remoteok.com/api` — returns JSON array, first element is metadata (skip it).

```typescript
import { Job, getCompanyLogoUrl } from "../types";
import { isWorldwideRemote } from "../filter";
import { categorizeJob } from "../categorize";
import { createJobId } from "../utils";
import { fetchWithRetry } from "../fetch-retry";

export interface RemoteOKResult {
  jobs: Job[];
  rawCount: number;
}

export async function scrapeRemoteOK(): Promise<RemoteOKResult> {
  console.log("[remoteok] starting API fetch...");

  const res = await fetchWithRetry("https://remoteok.com/api", {
    headers: {
      "User-Agent": "worldglide-jobs/1.0",
      Accept: "application/json",
    },
    timeoutMs: 30000,
  });

  if (!res.ok) {
    throw new Error(`remoteok API ${res.status}`);
  }

  const data = await res.json();
  // First element is metadata/legal notice, skip it
  const items = Array.isArray(data) ? data.slice(1) : [];
  const rawCount = items.length;
  const allJobs: Job[] = [];

  for (const item of items) {
    const location = item.location || "";
    const description = item.description || "";

    if (
      !isWorldwideRemote({
        title: item.position,
        description,
        location,
        tags: item.tags || [],
      })
    ) {
      continue;
    }

    const category = categorizeJob(item.position);
    if (!category) continue;

    const companyDomain = item.company_logo
      ? undefined // RemoteOK doesn't always expose domain
      : undefined;

    allJobs.push({
      id: createJobId("remoteok", `${item.id}`),
      title: item.position || "",
      company: item.company || "Unknown",
      companyLogo: item.company_logo || undefined,
      category,
      url: item.url || `https://remoteok.com/remote-jobs/${item.id}`,
      source: "remoteok",
      tags: item.tags || [],
      postedAt: item.date || new Date().toISOString(),
      scrapedAt: new Date().toISOString(),
      description: stripHtml(description).slice(0, 200),
      isWorldwide: true,
      employmentType: "Full-time",
    });
  }

  console.log(`[remoteok] ${allJobs.length} worldwide jobs from ${rawCount} raw`);
  return { jobs: allJobs, rawCount };
}

function stripHtml(html: string): string {
  return html
    .replace(/<[^>]*>/g, " ")
    .replace(/\s+/g, " ")
    .trim();
}
```

**Step 3: Build API aggregator orchestrator**

Create `src/lib/api-aggregator.ts`:

```typescript
import { Job } from "./types";
import { scrapeHimalayas } from "./scrapers/api-himalayas";
import { scrapeRemoteOK } from "./scrapers/api-remoteok";

export interface ApiAggregatorResult {
  jobs: Job[];
  rawTotal: number;
  sources: { name: string; jobs: number; raw: number; error?: string }[];
}

export async function scrapeExternalApis(): Promise<ApiAggregatorResult> {
  console.log("[api-aggregator] fetching external job APIs...");

  const scrapers = [
    { name: "himalayas", fn: scrapeHimalayas },
    { name: "remoteok", fn: scrapeRemoteOK },
  ];

  const allJobs: Job[] = [];
  let rawTotal = 0;
  const sources: ApiAggregatorResult["sources"] = [];

  // Run sequentially to respect rate limits
  for (const scraper of scrapers) {
    try {
      const result = await scraper.fn();
      allJobs.push(...result.jobs);
      rawTotal += result.rawCount;
      sources.push({ name: scraper.name, jobs: result.jobs.length, raw: result.rawCount });
    } catch (err) {
      const msg = err instanceof Error ? err.message : String(err);
      console.error(`[api-aggregator] ${scraper.name} failed: ${msg}`);
      sources.push({ name: scraper.name, jobs: 0, raw: 0, error: msg });
    }
  }

  console.log(`[api-aggregator] ${allJobs.length} worldwide jobs from ${rawTotal} raw across ${sources.length} APIs`);
  return { jobs: allJobs, rawTotal, sources };
}
```
